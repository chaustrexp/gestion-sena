document.addEventListener('DOMContentLoaded', function() {
    const toggleBtn = document.getElementById('toggleSidebar');
    const wrapper = document.getElementById('wrapper');
    const sidebar = document.querySelector('.sidebar');
    const body = document.body;
    
    // Detectar si es mÃ³vil
    const isMobile = () => window.innerWidth <= 768;
    
    // Cargar estado (solo desktop)
    if (!isMobile()) {
        const sidebarCollapsed = localStorage.getItem('sidebarCollapsed') === 'true';
        if (sidebarCollapsed) {
            wrapper.classList.add('sidebar-collapsed');
        }
    }
    
    // Toggle sidebar
    toggleBtn.addEventListener('click', function(e) {
        e.stopPropagation();
        
        if (isMobile()) {
            // Modo mÃ³vil: overlay
            wrapper.classList.toggle('sidebar-active');
            body.classList.toggle('sidebar-open');
        } else {
            // Modo desktop: collapse
            wrapper.classList.toggle('sidebar-collapsed');
            
            // Guardar estado
            const isCollapsed = wrapper.classList.contains('sidebar-collapsed');
            localStorage.setItem('sidebarCollapsed', isCollapsed);
        }
        
        // Reiniciar iconos
        setTimeout(() => {
            if (typeof lucide !== 'undefined') {
                lucide.createIcons();
            }
        }, 300);
    });
    
    // Cerrar en mÃ³vil al hacer clic fuera
    document.addEventListener('click', function(e) {
        if (isMobile() && wrapper.classList.contains('sidebar-active')) {
            if (!sidebar.contains(e.target) && !toggleBtn.contains(e.target)) {
                wrapper.classList.remove('sidebar-active');
                body.classList.remove('sidebar-open');
            }
        }
    });
    
    // Manejar resize
    let resizeTimer;
    window.addEventListener('resize', function() {
        clearTimeout(resizeTimer);
        resizeTimer = setTimeout(function() {
            if (isMobile()) {
                wrapper.classList.remove('sidebar-collapsed');
                body.classList.remove('sidebar-open');
            } else {
                wrapper.classList.remove('sidebar-active');
                body.classList.remove('sidebar-open');
                
                const sidebarCollapsed = localStorage.getItem('sidebarCollapsed') === 'true';
                if (sidebarCollapsed) {
                    wrapper.classList.add('sidebar-collapsed');
                }
            }
            
            if (typeof lucide !== 'undefined') {
                lucide.createIcons();
            }
        }, 250);
    });
});
